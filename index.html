import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load and analyze data
df = pd.read_csv('data.csv')
insights = df.groupby('category').agg({
    'revenue': ['sum', 'mean'],
    'users': 'count'
})

# Visualize results
insights.plot(kind='bar', figsize=(10,6))
plt.title('Data Insights Dashboard')
plt.show()
